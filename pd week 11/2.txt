void again_file_out()
{
	
	file.open("alphabets.txt",ios::out);
	for(char i='a' ; i<='z' ; i++)
	{
		for(int j=0 ; alph[j]!='\0' ; j++)
		{
			if(check(i,alph[j]))
			{
				
				file << alph[j];	
				break;
			}
			
			else
			{
				file << i;
				break;
			}	
		}	
		
	}
	file.close();
}
